// basic calculator
#include<stdio.h>
int main()
{
float a,b,c;
char 'o' ;
printf("Enter the operands a,b :");
scanf("%f %f",&a,&b);
switch(o)
{
case'+' :
         c = a+b ;
         break;
case'-' :
        c = a-b ;
        break;
case'*' :        
        c=a*b ;
        break;
case'/' :
       c=a/b ;
       break;
case'%' :
       c=a%b ;
       break;
  }     
printf("%f %c%f=%f",a,o,b,c);
return 0;
}
